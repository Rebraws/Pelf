128template <class Container>
129constexpr auto Pe<Container>::checkPESignature() const -> bool {
130 /* Read PE Signature from offset mPeHeaderAddress*/
131 std::uint32_t pe_signature{};
132 std::size_t offset = mPeHeaderAddress;
133
134 /* mPeSignature is set as 0x5045000, so no need to read the value backwards */
135 for (std::size_t i{}; i < sizeof(pe_signature); ++i) {
136 pe_signature <<= 8;
137 pe_signature |= this->mData[offset++];
138 }
139
140
141 return pe_signature == mPeSignature;

166template <class Container>
167constexpr auto Pe<Container>::readCoffHeader() -> void {
168
169 // coff header offset
170 auto offset = mPeHeaderAddress + 4;
171
172 if (std::is_constant_evaluated()){
173 PelfHelper::copyStruct(this->mData, mCoffHeader, offset);
174 } else {
175 assert(std::is_trivially_copyable<decltype(mCoffHeader)>::value);
176
177 if (mPeHeaderAddress + 4 + sizeof(mCoffHeader) < this->mData.size()) {
178 std::memcpy(reinterpret_cast<char *>(&mCoffHeader),
179 this->mData.data() + offset,
180 sizeof(mCoffHeader));
181 } else {
182 throw PelfException{"Bad PE file, invalid Coff Header"};
183 }
184 }
