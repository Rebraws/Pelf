38template<class Container>
39consteval auto getPeNumberOfSections(const Container& data) requires
40 container_and_convertible_v<Container, unsigned char>
41{
42
43 /* The number of sections it's at offset 6 from the Coff header , so first we
44 * need to read pe header address (at offset 0x3c) */
45 std::uint32_t pe_header_address{};
46
47 std::size_t offset{ 0x3c };
48
49
50 for (std::size_t i{}; i < sizeof(pe_header_address); ++i) {
51 pe_header_address <<= 8;
52 pe_header_address |= static_cast<std::uint32_t>(
53 data.at(sizeof(pe_header_address) - 1 + offset--));
54 }
55
56 /* Offset of the Number of Sections */
57 offset = pe_header_address + sizeof(DWORD) + sizeof(WORD);
58
59 WORD number_of_sections{};
60
61 for (std::size_t i{}; i < sizeof(number_of_sections); ++i) {
62 number_of_sections <<= 8;
63 number_of_sections |= data.at(sizeof(number_of_sections) - 1 + offset - i);
64 }
65
66 return number_of_sections;
67}

213template<class Container, std::size_t NumOfSections>
214constexpr auto Pe<Container, NumOfSections>::checkPESignature() const -> bool
215{
216 /* Read PE Signature from offset mPeHeaderAddress*/
217 std::uint32_t pe_signature{};
218 std::size_t offset = mPeHeaderAddress;
219
220 /* mPeSignature is set as 0x5045000, so no need to read the value backwards */
221 for (std::size_t i{}; i < sizeof(pe_signature); ++i) {
222 pe_signature <<= 8;
223 pe_signature |= this->mData.at(offset++);
224 }
225
226
227 return pe_signature == mPeSignature;

277template<class Container, std::size_t NumOfSections>
278constexpr auto Pe<Container, NumOfSections>::parseSections() -> void
279{
280
281 std::ptrdiff_t offset =
282 mPeHeaderAddress + sizeof(mHeaders.mCoffHeader)
283 + sizeof(mHeaders.mOptionalHeader.mDataDirectories) * 16
284 + sizeof(mHeaders.mOptionalHeader.mWsf)
285 + sizeof(mHeaders.mOptionalHeader.mScf);
286
287 if constexpr (!std::is_same_v<decltype(mSections),
288 std::vector<IMAGE_SECTION_HEADER>>) {
289
290 for (auto& section : mSections) {
291 this->readHeader(section, offset);
292 offset += sizeof(section);
293 }
294 } else {
295 for (int i{}; i < mHeaders.mCoffHeader.NumberOfSections; ++i) {
296 IMAGE_SECTION_HEADER tmp_section{};
297 this->readHeader(tmp_section, offset);
298 offset += sizeof(tmp_section);
299 mSections.push_back(tmp_section);
300 }
301 }
302}
