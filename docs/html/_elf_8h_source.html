176template<class Function>
177constexpr auto
178 Elf<Container, NumOfSections, NumOfProgHeaders>::getSectionTableSize(
179 const Elf64_Ehdr& elfHeader,
180 Function read_struct) -> std::size_t
181{
182 /* If the offset of the section table is zero, then the size has to be zero */
183 if (elfHeader.e_shoff == 0) { return 0; }
184
185 /* If e_shnum holds a value greater or equal than SHN_LORESERVE , then the
186 number of entries of the section table is in the variable sh_size which
187 is member of the inital entry in section header table
188 */
189 if (elfHeader.e_shnum >= SHN_LORESERVE) {
190 /* Read the initial entry in section header table*/
191 Elf64_Shdr section_header = {};
192 read_struct(section_header, elfHeader.e_shoff);
193 return section_header.sh_size;
194 }
195
196 return elfHeader.e_shnum;
197}

227template<class Container,
228 std::size_t NumOfSections,
229 std::size_t NumOfProgHeaders>
230constexpr auto
231 Elf<Container, NumOfSections, NumOfProgHeaders>::getElfTablesSize(
232 const Container& data)
233 -> TableSizes requires container_and_convertible_v<Container, unsigned char>
234{
235
236 /* Lambda function that takes a reference to a header as a parameter
237 and initializes its values by reading `data` */
238 auto read_struct = [&](auto& header, std::size_t offset) {
239 hana::for_each(hana::keys(header), [&](auto key) {
240 auto& member = hana::at_key(header, key);
241
242#pragma unroll
243 for (std::size_t i{}; i < sizeof(member); ++i) {
244 if (sizeof(member) > 1) { member <<= 8; }
245 member |= data.at(sizeof(member) - 1 + offset - i);
246 }
247
248 offset += sizeof(member);
249 });
250 };
251
252
253 TableSizes tables_size;
254
255 /* Need to read some headers in order to extract the section table size
256 and the program header table size */
257
258 Elf64_Ehdr elf_header = {};
259
260 /* Read the elf header */
261 read_struct(elf_header, mElfHeaderOffset);
262
263 tables_size.sectionTable = getSectionTableSize(elf_header, read_struct);
264
265 tables_size.programHeader = getProgramTableSize(elf_header, read_struct);
266
267
268 return tables_size;
269}

302template<class Container,
303 std::size_t NumOfSections,
304 std::size_t NumOfProgHeaders>
305constexpr auto Elf<Container, NumOfSections, NumOfProgHeaders>::parseHeaders()
306 -> void
307{
308
309 /* Read the elf header into mElfHeaders.elfHeader, the elf header is always at
310 * offset 0 */
311 this->readHeader(mHeaders.elfHeader, 0);
312
313
314 // e_phoff can't be negative, unless it's an invalid file and an exception
315 // will be thrown later
316 std::ptrdiff_t offset =
317 static_cast<std::ptrdiff_t>(mHeaders.elfHeader.e_phoff); // program header table offset
318
319 /* Read the program header table */
320
321 if constexpr (!std::is_same_v<decltype(mHeaders.programHeaders),
322 std::vector<Elf64_Phdr>>) {
323 /* If offset is zero the programHeader container will be empty */
324 for (auto& header : mHeaders.programHeaders) {
325 this->readHeader(header, offset);
326 offset += sizeof(header);
327 }
328 } else { /* Runtime case! */
329 /* Get number of entries in the program header table */
330 std::size_t program_header_table_size{};
331 if (mHeaders.elfHeader.e_phnum >= PN_XNUM) {
332 program_header_table_size = getElfTablesSize(this->mData).programHeader;
333 } else {
334 program_header_table_size = mHeaders.elfHeader.e_phnum;
335 }
336
337 for (std::size_t i{}; i < program_header_table_size; ++i) {
338 Elf64_Phdr tmp = {};
339 this->readHeader(tmp, offset);
340 mHeaders.programHeaders.push_back(tmp);
341 offset += sizeof(tmp);
342 }
343 }
344}
