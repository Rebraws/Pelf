199template<class Container,

201 class Derived,
202 std::size_t... N>
203template<class Header>
204constexpr auto Pelf<Container, Derived, N...>::readHeader(Header& header,
205 const std::ptrdiff_t offset) -> void
206{
207
208 if (offset < 0) {
209 throw PelfException{ "Invalid offset (negative), while parsing headers" };
210 }
211
212 if (std::is_constant_evaluated()) {
213 header = this->template getStruct<Header>(static_cast<std::size_t>(offset));
214 } else {
215
216 assert(std::is_trivially_copyable<Header>::value);
217
218 if (static_cast<std::size_t>(offset) + sizeof(header)
219 <= this->mData.size()) {
220 std::copy(this->mData.begin() + offset,
221 this->mData.begin() + offset + sizeof(header),
222 reinterpret_cast<char*>(&header));
223 } else {
224 throw PelfException{ "Invalid Header" };
225 }
226 }
227}

230template<class Container,
231 template<typename, std::size_t...>
232 class Derived,
233 std::size_t... N>
234template<class Struct>
235constexpr auto Pelf<Container, Derived, N...>::getStruct(std::size_t offset)
236 -> Struct
237{
238
239 Struct s{};
240
241 hana::for_each(hana::keys(s), [&](auto key) {
242 auto& member = hana::at_key(s, key);
243
244#pragma unroll
245 for (std::size_t i{}; i < sizeof(member); ++i) {
246 if (sizeof(member) > 1) { member <<= 8; }
247 member |= mData.at(sizeof(member) - 1 + offset - i);
248 }
249
250 offset += sizeof(member);
251 });
252
253 return s;
254}
